### 2. 进行任务调度
系统采用FreeRTOS实现多任务管理，所有系统工作流程相关的代码均位于 freertos.c 文件中，
实现了基于FreeRTOS的多任务调度。包括:
- Sensor_Task ：传感器数据采集（1秒周期）
  
  - 读取DHT11温湿度数据
  - 读取TCRT5000红外数据
  - 读取HX711重量数据
  - 读取DS1302实时时间
- Oled_Show ：OLED数据显示（500ms周期）
  
  - 显示温湿度、重量、红外数据
  - 显示当前时间
- ESP_SendTask ：数据上传云端（1秒周期）
  
  - 封装传感器数据为JSON格式
  - 通过MQTT协议发送到云端
- ESP_ReceiveTask ：云端指令接收（300ms周期）
  
  - 接收云端控制指令（如蜂鸣器开关、舵机控制）
  - 执行相应的控制操作
- alert_task ：报警与提醒（3秒周期）
  
  - 检测是否到达吃药时间
  - 控制蜂鸣器和语音模块进行吃药提醒
  - 控制舵机自动出药
- Key_Task ：按键处理（10ms周期）
  
  - 检测按键事件（短按、长按、双击）
  - 处理本地手动控制逻辑
  - 紧急停止舵机等操作

## 具体的传感器与单片机通信实现
### 1. DHT11温湿度传感器
功能用途 ：

- 测量药箱内部环境温度（范围：-20~50°C）
- 测量药箱内部环境湿度（范围：5%~95%）
- 确保药品存储在适宜的温湿度环境中
引脚连接 ：

- 数据引脚（DQ）： PB12
通信方式 ：

- 1-Wire单总线协议
- 通过高低电平脉冲的持续时间来传输数据
- 一次通信传输40位数据（湿度整数+湿度小数+温度整数+温度小数+校验位）
关键代码 ：
```c
/**
 * @brief       复位DHT11
 */
static void dht11_reset(void)
{
    DHT11_DQ_OUT(0);    /* 拉低DQ */
    delay_ms(20);       /* 拉低至少18ms */
    DHT11_DQ_OUT(1);    /* DQ=1 */
    delay_us(30);       /* 主机拉高10~35us */
}

/**
 * @brief       等待DHT11的回应
 * @retval      0, DHT11正常
 *              1, DHT11异常/不存在
 */
uint8_t dht11_check(void)
{
    uint8_t retry = 0;
    uint8_t rval = 0;

    while (DHT11_DQ_IN && retry < 100)  /* DHT11会拉低83us */
    {
        retry++;
        delay_us(1);
    }

    if (retry >= 100)
    {
        rval = 1;
    }
    else
    {
        retry = 0;
        while (!DHT11_DQ_IN && retry < 100) /* DHT11拉低后会再次拉高87us */
        {
            retry++;
            delay_us(1);
        }
        if (retry >= 100) rval = 1;
    }
    return rval;
}

/**
 * @brief       从DHT11读取一个位
 * @retval      读取到的位值: 0 / 1
 */
uint8_t dht11_read_bit(void)
{
    uint8_t retry = 0;
    while (DHT11_DQ_IN && retry < 100)  /* 等待变为低电平 */
    {
        retry++;
        delay_us(1);
    }
    retry = 0;
    while (!DHT11_DQ_IN && retry < 100) /* 等待变高电平 */
    {
        retry++;
        delay_us(1);
    }
    delay_us(40);       /* 等待40us */
    return DHT11_DQ_IN ? 1 : 0; /* 根据引脚状态返回 bit */
}

/**
 * @brief       从DHT11读取一个字节
 * @retval      读到的数据
 */
static uint8_t dht11_read_byte(void)
{
    uint8_t i, data = 0;
    for (i = 0; i < 8; i++)         /* 循环读取8位数据 */
    {
        data <<= 1;                 /* 高位数据先输出, 先左移一位 */
        data |= dht11_read_bit();   /* 读取1bit数据 */
    }
    return data;
}

/**
 * @brief       从DHT11读取一次数据
 * @param       data: 温湿度数据结构体指针
 * @retval      0, 正常. 1, 失败
 */
uint8_t dht11_read_data(struct dht11_Struct *data)
{
    uint8_t buf[5];
    uint8_t i;
    dht11_reset();

    if (dht11_check() == 0)
    {
        for (i = 0; i < 5; i++)     /* 读取40位数据 */
        {
            buf[i] = dht11_read_byte();
        }

        if ((buf[0] + buf[1] + buf[2] + buf[3]) == buf[4])
        {
            data->humi = buf[0];
            data->temp = buf[2];
        }
    }
    else
    {
        return 1;
    }
    return 0;
}

/**
 * @brief       初始化DHT11的IO口同时检测DHT11的存在
 * @retval      0, 正常. 1, 不存在/不正常
 */
uint8_t dht11_init(void)
{
    GPIO_InitTypeDef gpio_init_struct;
    DHT11_DQ_GPIO_CLK_ENABLE();     /* 开启DQ引脚时钟 */
    gpio_init_struct.Pin = DHT11_DQ_GPIO_PIN;
    gpio_init_struct.Mode = GPIO_MODE_OUTPUT_OD; /* 开漏输出 */
    gpio_init_struct.Pull = GPIO_PULLUP;         /* 上拉 */
    gpio_init_struct.Speed = GPIO_SPEED_FREQ_LOW;/* 高速 */
    HAL_GPIO_Init(DHT11_DQ_GPIO_PORT, &gpio_init_struct);
    dht11_reset();
    return dht11_check();
}
```
### 2. HX711称重传感器
功能用途 ：

- 测量药箱中药品的重量
- 可用于检测药品是否被取出或放入
- 通过重量变化判断药品使用情况
引脚连接 ：

- 时钟引脚（SCK）： PB10
- 数据引脚（DOUT）： PB11
通信方式 ：

- 2线串行接口（时钟+数据）
- 上升沿接收数据，24位数据格式
- 第25个时钟脉冲用于设置增益和通道选择
关键代码 ：
```c
uint32_t Weight_Maopi;

// 初始化HX711相关GPIO
void HX711_Init(void)
{
    GPIO_InitTypeDef GPIO_InitStruct = {0};
    HX711_GPIO_CLK_ENABLE();      /* 开启GPIO时钟 */
    
    // 配置SCK引脚为输出
    GPIO_InitStruct.Pin = HX711_SCK_Pin;
    GPIO_InitStruct.Mode = GPIO_MODE_OUTPUT_PP;
    GPIO_InitStruct.Pull = GPIO_NOPULL;
    GPIO_InitStruct.Speed = GPIO_SPEED_FREQ_HIGH;
    HAL_GPIO_Init(HX711_SCK_GPIO_Port, &GPIO_InitStruct);
    
    // 配置DOUT引脚为输入
    GPIO_InitStruct.Pin = HX711_DOUT_Pin;
    GPIO_InitStruct.Mode = GPIO_MODE_INPUT;
    GPIO_InitStruct.Pull = GPIO_NOPULL;
    HAL_GPIO_Init(HX711_DOUT_GPIO_Port, &GPIO_InitStruct);
    
    delay_ms(100);
    Weight_Maopi = HX711_Read();        //获取毛皮重量
}

// 读取HX711数据
unsigned long HX711_Read(void)
{
    uint32_t data = 0;
    uint32_t i;
    
    HAL_GPIO_WritePin(HX711_SCK_GPIO_Port, HX711_SCK_Pin, GPIO_PIN_RESET);
    while(HAL_GPIO_ReadPin(HX711_DOUT_GPIO_Port, HX711_DOUT_Pin) == GPIO_PIN_SET);
    
    for(i = 0; i < 24; i++)
    {
        HAL_GPIO_WritePin(HX711_SCK_GPIO_Port, HX711_SCK_Pin, GPIO_PIN_SET);
        data = data << 1;
        HAL_GPIO_WritePin(HX711_SCK_GPIO_Port, HX711_SCK_Pin, GPIO_PIN_RESET);
        if(HAL_GPIO_ReadPin(HX711_DOUT_GPIO_Port, HX711_DOUT_Pin) == GPIO_PIN_SET)
        {
            data++;
        }
    }
    
    HAL_GPIO_WritePin(HX711_SCK_GPIO_Port, HX711_SCK_Pin, GPIO_PIN_SET);
    data = data ^ 0x800000; // 第25个脉冲下降沿来时，转换数据
    HAL_GPIO_WritePin(HX711_SCK_GPIO_Port, HX711_SCK_Pin, GPIO_PIN_RESET);
    
    return data;
}

// 称重函数
int32_t Get_Weight(void)
{
    // 读取HX711传感器数据
    const int32_t currentReading = HX711_Read();
    
    // 初始化结果变量
    int32_t objectWeight = 0;
    
    // 仅当读取值大于皮重时计算重量
    if (currentReading > Weight_Maopi)
    {
        // 计算物体的原始AD差值
        const int32_t adcDifference = currentReading - Weight_Maopi;
        
        // 应用校准系数转换为实际重量
        objectWeight = (int32_t)((float)adcDifference / GapValue);
    }
    
    return objectWeight;
}
```
### 3. DS1302实时时钟
功能用途 ：

- 提供实时时间和日期信息
- 可设置定时开关药箱的功能
- 记录药品使用时间戳
引脚连接 ：

- 时钟引脚（SCLK）： PB5
- 使能引脚（CE/RST）： PB7
- 数据引脚（I/O）： PB6
通信方式 ：

- 3线串行接口（时钟+使能+数据）
- 数据引脚为双向I/O，需要动态切换输入/输出模式
- 先发送地址，再发送/接收数据
关键代码 ：
```c

TIMEData TimeData;

void DS1302_GPIO_Init(void) {

    __HAL_RCC_GPIOA_CLK_ENABLE();
    __HAL_RCC_GPIOB_CLK_ENABLE();

// 向DS1302写入一个字节
void DS1302_write_onebyte(uint8_t data) {
    uint8_t count = 0;
    SCLK_L;
    DS1302_DATAOUT_init();
    
    for (count = 0; count < 8; count++) {
        SCLK_L;
        if (data & 0x01) {
            DATA_H;
        } else {
            DATA_L;
        }
        SCLK_H;
        data >>= 1;
    }
}

// 向DS1302寄存器写入数据
void DS1302_wirte_rig(uint8_t address, uint8_t data) {
    uint8_t temp1 = address;
    uint8_t temp2 = data;
    
    CE_L;
    SCLK_L;
    delay_us(10);
    CE_H;
    delay_us(10);
    
    DS1302_write_onebyte(temp1);
    DS1302_write_onebyte(temp2);
    
    CE_L;
    SCLK_L;
    delay_us(10);
}

// 从DS1302寄存器读取数据
uint8_t DS1302_read_rig(uint8_t address) {
    uint8_t temp3 = address;
    uint8_t count = 0;
    uint8_t return_data = 0x00;
    
    CE_L;
    SCLK_L;
    delay_us(10);
    CE_H;
    delay_us(10);
    
    DS1302_write_onebyte(temp3);
    DS1302_DATAINPUT_init();
    delay_us(10);
    
    for (count = 0; count < 8; count++) {
        delay_us(10);
        return_data >>= 1;
        SCLK_H;
        delay_us(10);
        SCLK_L;
        delay_us(10);
        if (HAL_GPIO_ReadPin(DS1302_DATA_PORT, DS1302_DATA_PIN)) {
            return_data = return_data | 0x80;
        }
    }
    
    delay_us(10);
    CE_L;
    DATA_L;
    return return_data;
}

// 初始化DS1302时钟
void DS1302_Init(const TIMEData *time) {
    unsigned char sec_bcd = ((time->second / 10) << 4) | (time->second % 10);
    unsigned char min_bcd = ((time->minute / 10) << 4) | (time->minute % 10);
    unsigned char hour_bcd = ((time->hour / 10) << 4) | (time->hour % 10);
    unsigned char day_bcd = ((time->day / 10) << 4) | (time->day % 10);
    unsigned char month_bcd = ((time->month / 10) << 4) | (time->month % 10);
    unsigned char year_bcd = ((time->year - 2000) / 10) << 4 | ((time->year - 2000) % 10);
    
    DS1302_wirte_rig(0x8e, 0x00); // 关闭写保护
    DS1302_wirte_rig(0x80, sec_bcd); // seconds
    DS1302_wirte_rig(0x82, min_bcd); // minutes

void DS1302_read_realTime(void) {
    DS1302_read_time();
    TimeData.second = (read_time[0] >> 4) * 10 + (read_time[0] & 0x0f);
    TimeData.minute = (read_time[1] >> 4) * 10 + (read_time[1] & 0x0f);
    TimeData.hour = (read_time[2] >> 4) * 10 + (read_time[2] & 0x0f);
    TimeData.day = (read_time[3] >> 4) * 10 + (read_time[3] & 0x0f);
    TimeData.month = (read_time[4] >> 4) * 10 + (read_time[4] & 0x0f);
    TimeData.week = read_time[5];
    TimeData.year = (read_time[6] >> 4) * 10 + (read_time[6] & 0x0f) + 2000;
}
```

```c
void SYN_FrameInfo(uint8_t Music, const char *HZdataFormat, ...) {
    // 格式化输入文本
    char formattedData[100];
    va_list args;
    va_start(args, HZdataFormat);
    vsprintf(formattedData, HZdataFormat, args);
    va_end(args);
    
    // 计算数据长度
    unsigned char HZ_Length = strlen(formattedData);
    
    // 构造帧头和数据
    unsigned char Frame_Info[100];
    Frame_Info[0] = 0xFD;  // 帧头
    Frame_Info[1] = 0x00;  // 帧长度高字节
    Frame_Info[2] = HZ_Length + 3;  // 帧长度低字节
    Frame_Info[3] = 0x01;  // 命令字
    Frame_Info[4] = 0x01 | Music << 4;  // 命令参数
    
    // 计算校验码
    unsigned char ecc = 0;
    for (int i = 0; i < 5; i++) {
        ecc ^= Frame_Info[i];
    }
    for (int i = 0; i < HZ_Length; i++) {
        ecc ^= formattedData[i];
    }
    
    // 复制数据和校验码
    memcpy(&Frame_Info[5], formattedData, HZ_Length);
    Frame_Info[5 + HZ_Length] = ecc;
    
    // 发送帧信息
```

### 11. 系统集成

所有传感器的数据最终通过FreeRTOS的 `Sensor_Task` 任务进行统一采集和处理，然后通过OLED显示或ESP8266上传到云端。系统采用模块化设计，每个传感器都有独立的驱动文件，便于维护和扩展。

```c

uint32_t BEEP;
uint32_t SG90;

void Sensor_Task(void *argument){
    while(1){
        dht11_read_data(&DHT11_Data);               // 读取温湿度数据
        TCRT5000 = adc_get_result(ADC_CHANNEL_1);   // 读取红外传感器数据
        Weight = Get_Weight();                      // 读取重量数据

        DS1302_read_realTime();                     // 读取实时时间

        vTaskDelay(1000);                           // 任务周期1秒
    }
}
```
